CREATE KEYSPACE counterkeyspace
WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 1 };

CREATE TABLE counter_timeline(
	counter_id text,
	timestamp bigint,
	avg_sec_gap float,
	speed int,
	cars_per_sec int,
	PRIMARY KEY(counter_id, timestamp)
)
WITH CLUSTERING ORDER BY (timestamp DESC);
